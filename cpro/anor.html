#include<stdio.h>

int main()

{

int n,a,b,num,n1,i,d=0;

printf("Enter the starting range: ");

scanf("%d",&b);

printf("Enter the ending range: ");

scanf("%d",&n);

for(i=b;i<=n;i++)

{

d=0;

num=i;

n1=num;

while(num>0)

{

a=num%10;

d=d+a*a*a;

num=num/10;

}

if(d==n1)

{

printf("%d is Armstrong\n",n1);

}

}

}
