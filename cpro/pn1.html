14.PROGRAM T0 DISPLAY PRIME NUMBER WITHIN RANGE :

#include<stdio.h>

int main()

{

int n,i,j,a,b;

printf("Enter the starting range : ");

scanf("%d",&a);

printf("Enter the ending range : ");

scanf("%d",&b);

for(i=a;i<=b;i++)

{

n=0;

for(j=1;j<=b;j++)

{

if(i%j==0)

n++;

}

if(n==2)

printf("%d is prime number \n",i);

}

}
