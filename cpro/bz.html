29.PROGRAM T0 CHECK IF THE NUMBER IS BUZZ OR NOT : (A number is said to be Buzz Number if it ends with 7 OR is divisible by 7.)

#include<stdio.h>

int main()

{

int n,n1;

printf("Enter the number: ");

scanf("%d",&n);

n1=n;

if(n%7==0||n%10==7)

{

printf("%d is buzz number",n1);

}

else

{

printf("%d is not buzz number",n1);

}

}
