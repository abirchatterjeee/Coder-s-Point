25.PROGRAM T0 FIND AUTOMORFIC WITHIN RANGE:

#include<stdio.h>

int main()

{

int i,j,n,n1,z=0;

printf("Enter the start range:\n");

scanf("%d",&n);

printf("Enter the end range:\n");

scanf("%d",&n1);

for(i=n;i<=n1;i++)

{

j=i*i;

if(i>0&&i<=9)

{

z=j%10;

}

else if(i>=10&&i<100)

{

z=j%100;

}

else if(i>=100&&i<1000)

{

z=j%1000;

}

if(z==i)

{

printf("%d is automorphic\n",i);

}

}

}
