24.PROGRAM T0 CHECK IF THE NUMBER IS AUTOMORFIC OR NOT:

#include<stdio.h>

int main()

{

int i,j,z=0;

printf("Enter the number\n");

scanf("%d",&i);

j=i*i;

if(i>0&&i<=9)

{

z=j%10;

}

else if(i>=10&&i<100)

{

z=j%100;

}

else if(i>=100&&i<1000)

{

z=j%1000;

}

if(z==i)

{

printf("%d is automorphic\n",i);

}

else

{

printf("%d is not automorphic\n",i);

}

}

OUTPUT :

Enter the number

6
