20.PROGRAM T0 DISPLAY PALINDROM NUMBER WITHIN RANGE:

#include<stdio.h>

int main()

{

int n,b,a,num,n1,i,d=0;

printf("Enter the number ");

scanf("%d",&b);

printf("Enter the number ");

scanf("%d",&n);

for(i=b;i<=n;i++)

{

d=0;

num=i;

n1=num;

while(num>0)

{

a=num%10;

d=d*10+a;

num=num/10;

}

if(d==n1)

{

printf("%d is palindrome\n",n1);

}

}

}
