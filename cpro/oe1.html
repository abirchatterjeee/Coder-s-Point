9.PROGRAM TO DISPLAY THE EVEN OR ODD NUMBER WITHIN RANGE :

#include<stdio.h>

int main ()

{

int i,n;

printf("Enter the range:\n");

scanf("%d",&n);

for(i=1;i<=n;i++)

{

if(i%2==0)

{

printf("%d is even number\n",i);

}

else

{

printf("%d is odd number\n",i);

}

}

}

OUTPUT :

Enter the range :

5

1 is odd number

2 is even number

3 is odd number

4 is even number

5 is odd number
