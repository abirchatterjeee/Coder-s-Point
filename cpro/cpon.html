19.PROGRAM T0 CHECK IF THE NUMBER IS PALINDROM OR NOT:

#include<stdio.h>

int main()

{

int a,d=0,n,num;

printf("Enter the number ");

scanf("%d",&n);

num=n;

while(n)

{

a=n%10;

d=d*10+a;

n=n/10;

}

if(num==d)

{

printf("%d is palindrome",num);

}

else

{

printf("%d is not palindrome",num);

}

}

OUTPUT :

Enter the number 154

154 is not palindrome

Enter the number 454

454 is palindrome
