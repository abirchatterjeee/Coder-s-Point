28.PROGRAM T0 CHECK IF THE NUMBER I IS SPECIAL OR NOT :( A number is said to be special number when the sum of factorial of its digits is equal to the number itself. Example- 145 is a Special Number as 1!+4!+5!=145.)

#include<stdio.h>

int main()

{

int n,a,i,n1,s=0,f=1;

printf("Enter the number: ");

scanf("%d",&n);

n1=n;

while(n>0)

{

a=n%10;

f=1;

for(i=1;i<=a;i++)

{

f=f*i;

}

s=s+f;

n=n/10;

}

if(s==n1)

{

printf("%d is special number",n1);

}

else

{

printf("%d is not special number",n1);

}

}

OUTPUT :

Enter the number: 145

145 is special number
