13.PROGRAM TO CHECK WHETHER THE NUMBER IS PRIME OR NOT :

#include<stdio.h>

int main()

{

int i,n,c=0;

printf("Enter the number: ");

scanf("%d",&n);

for(i=1;i<=n;i++)

{

if(n%i==0)

{

c=c+1;

}

}

if(c<=2)

{

printf("%d is a prime number",n);

}

else

{

printf("%d is not a prime number",n);

}

}

OUTPUT :

Enter the number: 10

10 is not a prime number
