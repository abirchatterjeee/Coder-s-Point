21.PROGRAM T0 CHECK THE NUMBER IS ARMSTRONG OR NOT:

#include<stdio.h>

int main()

{

int n,a,num,d=0;

printf("Enter the number ");

scanf("%d",&n);

num=n;

while(n)

{

a=n%10;

d=d+a*a*a;

n=n/10;

}

if(num==d)

{

printf("%d is armstrong",num);

}

else

{

printf("%d is not armstrong",num);

}

}

Enter the number 153

Enter the number 458
