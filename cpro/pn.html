10.PROGRAM TO CHECK WHETHER THE NUMBER IS PERFECT OR NOT :

#include<stdio.h>

int main ()

{

int i,n,sum=0;

printf("Enter the number:");

scanf("%d",&n);

for(i=1;i<n;i++)

{

if(n%i==0)

{

sum=sum+i;

}

}

if(sum==n)

{

printf("%d is perfect number",i);

}

else

{

printf("%d is not perfect number",i);

}

}

OUTPUT :

Enter the number:45

45 is not perfect number
