2.PROGRAM TO SUBSTRUCTION OF TWO NUMBER.

#include<stdio.h>

int main()

{

int a,b,s=0;

printf("Enter the 1st num : ");

scanf("%d",&a);

printf("Enter the 2nd num : ");

scanf("%d",&b);

s=a-b;

printf("Substruction of two number is %d ",s);

return 0;

}

OUTPUT :

Enter the 1st num :

3

Enter the 2nd num :

2

Substruction of two num is 1
