3.PROGRAM TO MULTIPLY TWO NUMBERS.

#include<stdio.h>

int main()

{

int a,b,s=0;

printf("Enter the 1st num : ");

scanf("%d",&a);

printf("Enter the 2nd num : ");

scanf("%d",&b);

s=a*b;

printf("Product of two num is %d ",s);

return 0;

}

OUTPUT :

Enter the 1st num :

3

Enter the 2nd num :

2

Product of two num is 6
