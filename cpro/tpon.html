23.PROGRAM T0 CHECK IF THE NUMBER IS TWINPRIME OR NOT:

#include<stdio.h>

int main()

{

int i,j,n,n1,c=0,c1=0,diff;

printf("Enter the first number:");

scanf("%d",&n);

printf("Enter the second number:");

scanf("%d",&n1);

diff=n-n1;

for(i=1;i<=n;i++)

{

if(n%i==0)

{

c=c+1;

}

}

if(c<=2)

{

for(j=1;j<=n1;j++)

{

if(n1%j==0)

{

c1=c1+1;

}

}

if(c1<=2)

{

if(diff==2||diff==-2)

{

printf("Twin Prime");

}

else

{

printf("Not twin prime");

}

}

else

{

printf("%d is not prime",n1);

}

}

else

{

printf("%d is not prime",n);

}

}
