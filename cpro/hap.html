27.PROGRAM T0 CHECK IF THE NUMBER IS HAPPY OR NOT: (A number whose sum of the square of digits is ultimatly equal to 1 is a happy number. By ultimately we mean that starting with any positive number we continue the addition of square of digits till we get a single digit number (0 ~ 9))

#include<stdio.h>

int main()

{

int n,a,n1,s=0;

printf("Enter the number: ");

scanf("%d",&n);

n1=n;

s=n;

while(s>9)

{

s=0;

while(n>0)

{

a=n%10;

s=s+a*a;

n=n/10;

}

n=s;

}

if(s==1)

{

printf("%d is happy number",n1);

}

else

{

printf("%d is not happy number",n1);

}

}

OUTPUT :

Enter the number: 68

68 is happy number
