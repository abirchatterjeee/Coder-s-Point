15.PROGRAM T0 ENTER ANY NUMBER AND DISPLAY AFTER EXTRACTION:

#include<stdio.h>

int main()

{

int n,a,b,d=0;

printf("Enter the number: ");

scanf("%d",&n);

printf("Digits after extraction are:\n");

while(n>0)

{

a=n%10;

d=d*10+a;

n=n/10;

}

while(d>0)

{

b=d%10;

d=d/10;

printf("%d ",b);

}

}

OUTPUT :

Enter the number: 456

Digits after extraction are:

4 5 6
